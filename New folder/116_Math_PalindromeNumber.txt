public class Math_PalindromeNumber {
    public static void main(String[] args) {
        int num = 12321, original = num, reversed = 0;
        while (num != 0) {
            reversed = reversed * 10 + num % 10;
            num /= 10;
        }
        System.out.println(original + " is palindrome? " + (original == reversed));
    }
}
