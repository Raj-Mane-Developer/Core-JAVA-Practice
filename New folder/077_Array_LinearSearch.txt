public class Array_LinearSearch {
    public static void main(String[] args) {
        int[] arr = {5, 3, 8, 1, 9, 2};
        int target = 9;
        int index = -1;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                index = i;
                break;
            }
        }
        System.out.println(index != -1 ? "Found at index " + index : "Not found");
    }
}
