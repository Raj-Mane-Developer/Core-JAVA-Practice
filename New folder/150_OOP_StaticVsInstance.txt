public class OOP_StaticVsInstance {
    static int staticCount = 0;
    int instanceCount = 0;

    OOP_StaticVsInstance() {
        staticCount++;
        instanceCount++;
    }

    public static void main(String[] args) {
        OOP_StaticVsInstance obj1 = new OOP_StaticVsInstance();
        OOP_StaticVsInstance obj2 = new OOP_StaticVsInstance();
        System.out.println("Static count: " + staticCount);
        System.out.println("obj1 instance count: " + obj1.instanceCount);
        System.out.println("obj2 instance count: " + obj2.instanceCount);
    }
}
