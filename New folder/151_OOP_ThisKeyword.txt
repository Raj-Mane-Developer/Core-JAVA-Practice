public class OOP_ThisKeyword {
    public static void main(String[] args) {
        Point p = new Point(3, 4);
        p.display();
    }
}

class Point {
    int x, y;

    Point(int x, int y) {
        this.x = x;
        this.y = y;
    }

    void display() {
        System.out.println("Point(" + this.x + ", " + this.y + ")");
    }
}
