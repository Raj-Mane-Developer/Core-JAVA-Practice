public class Control_NestedIfEligibility {
    public static void main(String[] args) {
        int age = 20;
        boolean hasId = true;
        if (age >= 18) {
            if (hasId) {
                System.out.println("Eligible to vote");
            } else {
                System.out.println("Need valid ID to vote");
            }
        } else {
            System.out.println("Not eligible to vote");
        }
    }
}
