public class Sort_LinearSearchAllOccurrences {
    public static void main(String[] args) {
        int[] arr = {1, 3, 5, 3, 7, 3, 9};
        int target = 3;
        System.out.print("Found at indices: ");
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                System.out.print(i + " ");
            }
        }
        System.out.println();
    }
}
