public class Math_GCDUsingLoop {
    public static void main(String[] args) {
        int a = 48, b = 18, gcd = 1;
        for (int i = 1; i <= Math.min(a, b); i++) {
            if (a % i == 0 && b % i == 0) gcd = i;
        }
        System.out.println("GCD of " + a + " and " + b + " = " + gcd);
    }
}
