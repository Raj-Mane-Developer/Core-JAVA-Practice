public class Pattern_PascalsTriangle {
    public static void main(String[] args) {
        int n = 6;
        for (int line = 0; line < n; line++) {
            int val = 1;
            for (int s = 0; s < n - line; s++) System.out.print(" ");
            for (int i = 0; i <= line; i++) {
                System.out.print(val + " ");
                val = val * (line - i) / (i + 1);
            }
            System.out.println();
        }
    }
}
