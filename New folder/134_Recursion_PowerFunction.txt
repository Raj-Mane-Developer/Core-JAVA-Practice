public class Recursion_PowerFunction {
    public static void main(String[] args) {
        int base = 2, exp = 10;
        System.out.println(base + "^" + exp + " = " + power(base, exp));
    }

    static long power(int base, int exp) {
        if (exp == 0) return 1;
        return base * power(base, exp - 1);
    }
}
