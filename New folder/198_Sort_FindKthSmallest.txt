import java.util.Arrays;

public class Sort_FindKthSmallest {
    public static void main(String[] args) {
        int[] arr = {7, 10, 4, 3, 20, 15};
        int k = 3;
        int[] sorted = arr.clone();
        Arrays.sort(sorted);
        System.out.println(k + "th smallest element: " + sorted[k - 1]);
    }
}
